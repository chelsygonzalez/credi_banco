CREATE DATABASE tienda;
USE tienda;


CREATE TABLE usuario (correo_usu varchar(40) NOT NULL, id_usuario int(20) DEFAULT NULL, nomb_usu varchar(30) NOT NULL DEFAULT '',
apell_usu varchar(30) DEFAULT NULL, cel_usu varchar(11) DEFAULT NULL, contra_usu varchar(40) NOT NULL,
saldo_usu decimal(18, 0) unsigned DEFAULT '0.00',
foto_usu longblob, PRIMARY KEY (correo_usu)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ======================================================================================== --

CREATE TABLE empleado (correo_emp varchar(40) NOT NULL, nom_emple varchar(25) NOT NULL, 
apell_emp varchar(25) DEFAULT NULL, cel_emple int(10) DEFAULT NULL, id_centro1 varchar(20) DEFAULT NULL,
contra_emp varchar(30) DEFAULT NULL, cedu_emp int(20) DEFAULT NULL,
PRIMARY KEY (correo_emp)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE transacciones (id_tx int(10) auto_increment, valor_tx decimal(18, 0) unsigned DEFAULT NULL,
correo_emp1 varchar(40) DEFAULT NULL, fecha_tx datetime DEFAULT NULL, correo_usu1 varchar(40) DEFAULT NULL,
FOREIGN KEY (correo_emp1) references empleado(correo_emp), FOREIGN KEY (correo_usu1) references usuario(correo_usu),
PRIMARY KEY (id_tx)) 
ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


CREATE TABLE recargas (id_recar int(10) AUTO_INCREMENT, valor_recar decimal(18, 0) unsigned DEFAULT NULL,
correo_usu2 varchar(40) DEFAULT NULL, fecha_recar datetime DEFAULT NULL, 
FOREIGN KEY (correo_usu2) references usuario(correo_usu), PRIMARY KEY (id_recar)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
