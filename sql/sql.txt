CREATE DATABASE tienda;
USE tienda;

CREATE TABLE cen_formacion (id_centro VARCHAR(20) NOT NULL, nom_centro VARCHAR(50) DEFAULT NULL,
reg_centro VARCHAR(30) DEFAULT NULL, PRIMARY KEY (id_centro)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE usuario (correo_usu varchar(40) NOT NULL, id_usuario int(20) DEFAULT NULL, nomb_usu varchar(30) DEFAULT NULL,
apell_usu varchar(30) DEFAULT NULL, cel_usu int(30) DEFAULT NULL, contra_usu varchar(40) DEFAULT NULL,
foto_usu longblob, PRIMARY KEY (correo_usu)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ======================================================================================== --

CREATE TABLE empleado (correo_emp varchar(40) NOT NULL, nom_emple varchar(25) DEFAULT NULL, 
apell_emp varchar(25) DEFAULT NULL, cel_emple int(10) DEFAULT NULL, id_centro1 varchar(20) DEFAULT NULL,
contra_emp varchar(30) DEFAULT NULL, cedu_emp int(20) DEFAULT NULL, FOREIGN KEY (id_centro1) references cen_formacion(id_centro),
PRIMARY KEY (correo_emp)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE transacciones (id_tx int(10) NOT NULL, valor_tx int(20) DEFAULT NULL,
correo_emp1 varchar(40) DEFAULT NULL, fecha_tx date DEFAULT NULL, correo_usu1 varchar(40) DEFAULT NULL,
FOREIGN KEY (correo_emp1) references empleado(correo_emp), FOREIGN KEY (correo_usu1) references usuario(correo_usu),
PRIMARY KEY (id_tx)) 
ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


CREATE TABLE recargas (id_recar int(20) NOT NULL, valor_recar int(20) DEFAULT NULL,
correo_usu2 varchar(40) DEFAULT NULL, fecha_recar date DEFAULT NULL, 
FOREIGN KEY (correo_usu2) references usuario(correo_usu), PRIMARY KEY (id_recar)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
